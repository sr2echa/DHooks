### To use @everyone you can create announces via given example

```json
{
    "content": "@everyone",
    "embeds": [
        {
            "color": 7025265,
            "footer":
            {
                "text": "Official Announcement from your-server or discord channel"
                },
                "thumbnail":
                {
                     "url": "https://cdn.discordapp.com/attachments/304127205812731905/685806798103511073/iconfinder_Announcement_campgn_marketing_megaphone_promotion_sales_1886370.png"
                     },
                      "author":
                      {
                           "name": "ANNOUNCEMENT"
                           },
                            "fields":
                            [
                                {
                                     "name": "Announcement Number: **{{ColumnB}}**",
                                     "value": "{{ColumnC}}"
                                }
                            ]
                      }
                ]
}
````
